Loops:-
  for-loop
  syntax- for(initialisation ; condition ; updation)
          {
            //loop logic
          }

  while loop    
  syntax- //initialisation  
          while(condition)             
          {
            loop logic

            //updation
          }

  do-while loop
  syntax- //initialisation
          do
          {
            loop logic

            //updation
          }
          while(condition)
